Quick Start 
# Build the ISO
make build-iso

# Write to USB
USB_DEVICE=/dev/disk4 make write-usb

# Boot, login (root/kcore), run:
install-to-disk

# After reboot, manage VMs
kctl create vm web-server --cpu 4 --memory 8G
kctl get vms
